Modules

File location: \Steam\steamapps\common\Team Fortress 2\tf\cfg\overrides\modules.cfg 

// masterconfig settings
match_hud=off
jigglebones=off
hud_player_model=off
hud_achievement=off
htmlmotd=off
logo=off
props=low
ropes=low
sprays=on
gibs=off
sillygibs=off
ragdolls=off
characters=low
hud_avatars=off
killstreaks=off
mod_support=off
sound=ultra
voice_chat=off
motion_blur=off
particles=low
water=medium
shadows=off
flashlight=off
effects=medium
lighting_ex=high
lighting=medium_high
lod=high
anti_aliasing=msaa_8x

// spray fix
cl_spraydisable 0        // enable sprays
cl_allowdownload 1       // allows you to download other player's sprays
cl_allowupload 1         // uploads your sprays to other players
r_spray_lifetime 2       // keep sprays for 2 rounds
mp_decals 200            // decal limit
r_decals 2048            // decal limit
r_decal_cover_count 4
r_decal_cullsize 1 
r_decal_overlap_area 0.4 
r_decal_overlap_count 3 
r_decalstaticprops 1 

// fov
fov_desired 90

// mouse
m_rawinput 1
sensitivity 3.5

// gameplay
tf_medigun_autoheal 1
tf_sniper_fullcharge_bell 1
cl_autoreload 1
hud_combattext 1

hidepanel specgui

tf_dingalingaling 1
tf_dingalingaling_lasthit 1
tf_dingaling_volume 1
tf_dingaling_lasthit_volume 1

tf_dingaling_pitchmaxdmg 50
tf_dingaling_pitchmindmg 150

// hud
tf_contract_progress_show 0
cl_hud_minmode 0
cl_notifications_show_ingame 0
hud_classautokill 0
cl_spec_carrieditems 0
tf_scoreboard_ping_as_text 1
cl_disablehtmlmotd 1
tf_scoreboard_mouse_mode 2

// movement
unbind "w"
unbind "s"
unbind "a"
unbind "d"
unbind "q"
bind "w" "+forward"
bind "s" "+back"
bind "a" "+moveleft"
bind "d" "+moveright"
bind "q" "lastinv"

// alt+tab and sound restart
unbind "F9"
alias altabfix "record bugfix; wait 15; stop; snd_restart;"
bind "F9" "altabfix"

//Sentry Quickbuild Bind
unbind "MOUSE3"
bind "MOUSE3" "destroy 2 0; build 2 0"

// Discpenser Quickbuild Bind
unbind "MOUSE5"
bind "MOUSE5" "destroy 0 0; build 0 0"

// weapon model invisible toggle
unbind "v"
bind "v" "toggle r_drawviewmodel"

// net graph toggle
unbind "-"
bind "-" "toggle net_graph"

// Quick-Switch Loadouts
unbind "h"
unbind "j"
unbind "k"
unbind "l"
bind "h" "load_itempreset 0" 
bind "j" "load_itempreset 1"
bind "k" "load_itempreset 2"
bind "l" "load_itempreset 3"

// viewmodels big and small
unbind "["
bind "[" "toggle tf_use_min_viewmodels"

// viewmodels close and far
unbind "o"
unbind "p"
bind "o" "viewmodel_fov 70"
bind "p" "viewmodel_fov 85"

echo "MODULES ARE ENABLED!"
